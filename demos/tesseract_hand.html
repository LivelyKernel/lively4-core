Can we user Tesseract OCR for Handwriting reconition?

import Tesseract from 'https://cdn.rawgit.com/naptha/tesseract.js/1.0.10/dist/tesseract.js' var paper = this.parentElement.querySelector("#paper") paper.save = async () => { var image = paper.get("#canvas").getContext('2d') var result = await Tesseract.recognize(image) this.parentElement.querySelector("#result").innerHTML = result.html }
